CREATE USER STUD IDENTIFIED BY PASSWORD;

GRANT INSERT ON RegisteredFor (STUDID, CRN) TO STUD;



GRANT DELETE ON RegisteredFor (STUDID, CRN) TO STUD;


GRANT SELECT ON COURSE (CRN, DEPTKEY, SECTIONNUM, PROFNAME, DESC) TO STUD;

GRANT SELECT ON REGISTEREDFOR (COURSENUM, GRADE) TO STUD;


CREATE VIEW STUDENTGRADE AS 
	SELECT (COURSENUM, GRADE) FROM REGISTREDFOR WHERE STUDID = :OLD.STUDID;

GRANT SELECT ON STUDENTGRADE TO STUD;
