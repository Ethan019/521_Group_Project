DELIMITER $$
CREATE PROCEDURE LetterAssign(
	IN GRADE TINYINT(3),
	IN STUDID INT(7),
	OUT LETTERGRADE VARCHAR(1)
	)
BEGIN
    DECLARE gr TINYINT DEFAULT 0;
    SELECT GRADE INTO gr FROM STUDENTRECORD WHERE STUDID = '"+STUDID+"';
    IF gr >= 90 THEN
    	UPDATE STUDENTRECORD SET LETTERGRADE = "A" WHERE STUDID = '"+STUDID+"';
    ELSEIF gr >= 80 THEN
        UPDATE STUDENTRECORD SET LETTERGRADE = "B" WHERE STUDID = '"+STUDID+"';
    ELSEIF gr >= 70 THEN
    	UPDATE STUDENTRECORD SET LETTERGRADE = "c" WHERE STUDID = '"+STUDID+"';
    ELSEIF gr >= 60 THEN
    	UPDATE STUDENTRECORD SET LETTERGRADE = "D" WHERE STUDID = '"+STUDID+"'; 
    ELSE
        UPDATE STUDENTRECORD SET LETTERGRADE = "F" WHERE STUDID = '"+STUDID+"';
    END IF;
END$$
DELIMITER ;
